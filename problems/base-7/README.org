Given an integer, return its base 7 string representation.

Example 1:
#+BEGIN_EXAMPLE
Input: 100
Output: "202"
#+END_EXAMPLE

Example 2:
#+BEGIN_EXAMPLE
Input: -7
Output: "-10"
#+END_EXAMPLE
Note: The input will be in range of [-1e7, 1e7].

class Solution(object):
    def convertToBase7(self, num):
        """
        :type num: int
        :rtype: str
        """
        ## Idea:
        ## Complexity:
        if num == 0:
            return "0"

        is_positive = True
        if num < 0:
            is_positive = False
            num = -num

        res = ""
        while num != 0:
            res = "%s%s" % (str(num % 7), res)
            num = num/7
        return res if is_positive else "-" + res
