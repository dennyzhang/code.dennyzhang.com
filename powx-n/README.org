## Blog link: https://code.dennyzhang.com/powx-n
## Basic Ideas: 
##        If n < 0, x^n == (1/x)^(-n)
##        If n%2 == 0, x^n == (x*x)^(n/2)
##        If n%2 == 1, x^n == x * (x*x)^((n-1)/2)
##
## Complexity: Time O(log(n)), Space O(1)
class Solution(object):
    def myPow(self, x, n):
        """
        :type x: float
        :type n: int
        :rtype: float
        """
        if n == 0:
            return 1
        if n < 0:
            n = -n
            x = 1/x
        if n %2 == 0:
            return self.myPow(x*x, n/2)
        else:
            return x*self.myPow(x*x, (n-1)/2)
#+END_SRC
